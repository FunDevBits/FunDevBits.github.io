7 Insert Document

db.posts.insertOne({
  title: 'Post One',
  body: 'Body of post one',
  category: 'News',
  tags: ['news', 'events'],
  user: {
    name: 'John Doe',
    status: 'author'
  },
  date: Date()
})

8 Insert Multiple Documents

db.posts.insertMany([
  {
    title: 'Post Two',
    body: 'Body of post two',
    category: 'Technology',
    date: Date()
  },
  {
    title: 'Post Three',
    body: 'Body of post three',
    category: 'News',
    date: Date()
  },
  {
    title: 'Post Four',
    body: 'Body of post three',
    category: 'Entertainment',
    date: Date()
  }
])

18 Find Specific Fields

db.posts.find({ title: 'Post One' }, {
  title: 1,
  author: 1
})

19 Update Document

db.posts.updateOne({ title: 'Post Two' },
{
  $set: {
    body: 'Body for post 2',
    category: 'Technology'
  }
})

20 Update Multiple Document

db.posts.updateMany({ title: 'Post Two' },
{
  $set: {
    body: 'Body for post 2',
    category: 'Technology'
  }
})

21 Increment Field ($inc)

db.posts.update({ title: 'Post Two' },
{
  $inc: {
    likes: 5
  }
})

22 Rename Field

db.posts.update({ title: 'Post Two' },
{
  $rename: {
    likes: 'views'
  }
})

23 Delete Document

db.posts.remove({ title: 'Post Four' })

24 Sub-Documents

db.posts.update({ title: 'Post One' },
{
  $set: {
    comments: [
      {
	body: 'Comment One',
	user: 'Mary Williams',
	date: Date()
      },
      {
	body: 'Comment Two',
	user: 'Harry White',
	date: Date()
      }
    ]
  }
})

25 Find By Element in Array ($elemMatch)

db.posts.find({
  comments: {
     $elemMatch: {
       user: 'Mary Williams'
       }
    }
  }
)

26 Add Index

db.posts.createIndex({ title: 'text' })

27 Text Search (Single Word)

db.posts.find({
  $text: {
    $search: "Post"
    }
})

28 Text Search (Match Any of the Search Terms)

db.posts.find({
  $text: {
    $search: "Post One Two"
    }
})

29 Text Search (a Phrase)

db.posts.find({
  $text: {
    $search: "\"Post One\""
    }
})

30 Greater & Less Than

db.posts.find({ views: { $gt: 2 } })
db.posts.find({ views: { $gte: 7 } })
db.posts.find({ views: { $lt: 7 } })
db.posts.find({ views: { $lte: 7 } })
